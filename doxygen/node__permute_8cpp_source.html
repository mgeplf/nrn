9 /*
10 Below, the sense of permutation, is reversed. Though consistent, forward
11 permutation should be defined as (and the code should eventually transformed)
12 so that
13  v: original vector
14  p: forward permutation
15  pv: permuted vector
16  pv[i] = v[p[i]]
17 and
18  pinv: inverse permutation
19  pv[pinv[i]] = v[i]
20 Note: pinv[p[i]] = i = p[pinv[i]]
21 */
22 
23 /*
24 Permute nodes.
25 
26 To make gaussian elimination on gpu more efficient.
27 
28 Permutation vector p[i] applied to a data vector, moves the data_original[i]
29 to data[p[i]].
30 That suffices for node properties such as area[i], a[i], b[i]. e.g.
31  area[p[i]] <- area_original[i]
32 
33 Notice that p on the left side is a forward permutation. On the right side
34 it serves as the inverse permutation.
35 area_original[i] <- area_permuted[p[i]]
36 
37 but things
38 get a bit more complicated when the data is an integer index into the
39 original data.
40 
41 For example:
42 
43 parent[i] needs to be transformed so that
44 parent[p[i]] <- p[parent_original[i]] except that if parent_original[j] = -1
45  then parent[p[j]] = -1
46 
47 membrane mechanism nodelist ( a subset of nodes) needs to be at least
48 minimally transformed so that
49 nodelist_new[k] <- p[nodelist_original[k]]
50 This does not affect the order of the membrane mechanism property data.
51 
52 However, computation is more efficient to permute (sort) nodelist_new so that
53 it follows as much as possible the permuted node ordering, ie in increasing
54 node order. Consider this further mechanism specific nodelist permutation,
55 which is to be applied to the above nodelist_new, to be p_m, which has the same
56 size as nodelist. ie.
57 nodelist[p_m[k]] <- nodelist_new[k].
58 
59 Notice the similarity to the parent case...
60 nodelist[p_m[k]] = p[nodelist_original[k]]
61 
62 and now the membrane mechanism node data, does need to be permuted to have an
63 order consistent with the new nodelist. Since there are nm instances of the
64 mechanism each with sz data values (consider AoS layout).
65 The data permutation is
66 for k=[0:nm] for isz=[0:sz]
67  data_m[p_m[k]*sz + isz] = data_m_original[k*sz + isz]
68 
69 For an SoA layout the indexing is k + isz*nm (where nm may include padding).
70 
71 A more complicated case is a mechanisms dparam array (nm instances each with
72 dsz values) Some of those values are indices into another mechanism (eg
73 pointers to ion properties) or voltage or area depending on the semantics of
74 the value. We can use the above data_m permutation but then need to update
75 the values according to the permutation of the object the value indexes into.
76 Consider the permutation of the target object to be p_t . Then a value
77 iold = pdata_m(k, isz) - data_t in AoS format
78 refers to k_t = iold % sz_t and isz_t = iold - k_t*sz_t
79 and for a target in SoA format isz_t = iold % nm_t and k_t = iold - isz_t*nm_t
80 ie k_t_new = p_m_t[k_t] so, for AoS, inew = k_t_new*sz_t + isz_t
81 or , for SoA, inew = k_t_new + isz_t*nm_t
82 so pdata_m(k, isz) = inew + data_t
83 
84 
85 */

122 template <typename T>
123 void permute(T* data, int cnt, int sz, int layout, int* p) {
124  // data(p[icnt], isz) <- data(icnt, isz)
125  // this does not change data, merely permutes it.
126  // assert len(p) == cnt
127  if (!p) {
128  return;
129  }
130  int n = cnt * sz;
131  if (n < 1) {
132  return;
133  }
134 
135 #if CORENRN_BUILD
136  if (layout == Layout::SoA) { // for SoA, n might be larger due to cnt padding
137  n = nrn_soa_padded_size(cnt, layout) * sz;
138  }
139 #endif
140 
141  T* data_orig = new T[n];
142  for (int i = 0; i < n; ++i) {
143  data_orig[i] = data[i];
144  }
145 
146  for (int icnt = 0; icnt < cnt; ++icnt) {
147  for (int isz = 0; isz < sz; ++isz) {
148  // note that when layout==0, nrn_i_layout takes into account SoA padding.
149  int i = nrn_i_layout(icnt, cnt, isz, sz, layout);
150  int ip = nrn_i_layout(p[icnt], cnt, isz, sz, layout);
151  data[ip] = data_orig[i];
152  }
153  }
154 
155  delete[] data_orig;
156 }

190 static int full_search(NrnThread& nt, double* pd) {
191  int type = -1;
192  for (NrnThreadMembList* tml = nt.tml; tml; tml = tml->next) {
193  Memb_list* ml = tml->ml;
194  int n = corenrn.get_prop_param_size()[tml->index] * ml->_nodecount_padded;
195  if (pd >= ml->data && pd < ml->data + n) {
196  type = tml->index;
197  // insert into type_hints
198  int i = 0;
199  for (int type_hint: type_hints) {
200  if (type < type_hint) {
201  break;
202  }
203  i++;
204  }
205  type_hints.insert(type_hints.begin() + i, type);
206  break;
207  }
208  }
209  assert(type > 0);
210  return type;
211 }

214 int type_of_ntdata(NrnThread& nt, int i, bool reset) {
215  double* pd = nt._data + i;
216  assert(pd >= nt._actual_v);
217  if (pd < nt._actual_area) { // voltage first (area just after voltage)
218  return voltage;
219  }
220  assert(size_t(i) < nt._ndata);
221  // then check the type hints. When inserting a hint, keep in type order
222  if (reset) {
223  type_hints.clear();
224  }
225  for (int type: type_hints) {
226  Memb_list* ml = nt._ml_list[type];
227  if (pd >= ml->data) { // this or later
228  int n = corenrn.get_prop_param_size()[type] * ml->_nodecount_padded;
229  if (pd < ml->data + n) { // this is the one
230  return type;
231  }
232  } else { // earlier
233  return full_search(nt, pd);
234  }
235  }

237  return full_search(nt, pd);

242 static void update_pdata_values(Memb_list* ml, int type, NrnThread& nt) {
243  // assumes AoS to SoA transformation already made since we are using
244  // nrn_i_layout to determine indices into both ml->pdata and into target data
245  int psz = corenrn.get_prop_dparam_size()[type];
246  if (psz == 0) {
247  return;
248  }
249  if (corenrn.get_is_artificial()[type]) {
250  return;
251  }
252  int* semantics = corenrn.get_memb_func(type).dparam_semantics;
253  if (!semantics) {
254  return;
255  }
256  int* pdata = ml->pdata;
257  int layout = corenrn.get_mech_data_layout()[type];
258  int cnt = ml->nodecount;
259  // ml padding does not matter (but target padding does matter)
260 
261  // interesting semantics are -1 (area), -5 (pointer), -9 (diam), or 0-999 (ion variables)
262  for (int i = 0; i < psz; ++i) {
263  int s = semantics[i];
264  if (s == -1) { // area
265  int area0 = nt._actual_area - nt._data; // includes padding if relevant
266  int* p_target = nt._permute;
267  for (int iml = 0; iml < cnt; ++iml) {
268  int* pd = pdata + nrn_i_layout(iml, cnt, i, psz, layout);
269  // *pd is the original integer into nt._data . Needs to be replaced
270  // by the permuted value
271 
272  // This is ok whether or not area changed by padding?
273  // since old *pd updated appropriately by earlier AoS to SoA
274  // transformation
275  int ix = *pd - area0; // original integer into area array.
276  nrn_assert((ix >= 0) && (ix < nt.end));
277  int ixnew = p_target[ix];
278  *pd = ixnew + area0;
279  }
280  } else if (s == -9) { // diam
281  int diam0 = nt._actual_diam - nt._data; // includes padding if relevant
282  int* p_target = nt._permute;
283  for (int iml = 0; iml < cnt; ++iml) {
284  int* pd = pdata + nrn_i_layout(iml, cnt, i, psz, layout);
285  // *pd is the original integer into nt._data . Needs to be replaced
286  // by the permuted value
287 
288  // This is ok whether or not diam changed by padding?
289  // since old *pd updated appropriately by earlier AoS to SoA
290  // transformation
291  int ix = *pd - diam0; // original integer into actual_diam array.
292  nrn_assert((ix >= 0) && (ix < nt.end));
293  int ixnew = p_target[ix];
294  *pd = ixnew + diam0;
295  }
296  } else if (s == -5) { // POINTER
297  // assume pointer into nt._data. Most likely voltage.
298  // If not voltage, most likely same mechanism for all indices.
299  for (int iml = 0; iml < cnt; ++iml) {
300  int* pd = pdata + nrn_i_layout(iml, cnt, i, psz, layout);
301  int etype = type_of_ntdata(nt, *pd, iml == 0);
302  if (etype == voltage) {
303  int v0 = nt._actual_v - nt._data;
304  int* e_target = nt._permute;
305  int ix = *pd - v0; // original integer into area array.
306  nrn_assert((ix >= 0) && (ix < nt.end));
307  int ixnew = e_target[ix];
308  *pd = ixnew + v0;
309  } else if (etype > 0) {
310  // about same as for ion below but check each instance
311  Memb_list* eml = nt._ml_list[etype];
312  int edata0 = eml->data - nt._data;
313  int ecnt = eml->nodecount;
314  int esz = corenrn.get_prop_param_size()[etype];
315  int elayout = corenrn.get_mech_data_layout()[etype];
316  int* e_permute = eml->_permute;
317  int i_ecnt, i_esz, padded_ecnt;
318  int ix = *pd - edata0;
319  if (elayout == Layout::AoS) {
320  padded_ecnt = ecnt;
321  i_ecnt = ix / esz;
322  i_esz = ix % esz;
323  } else { // SoA
324  assert(elayout == Layout::SoA);
325  padded_ecnt = nrn_soa_padded_size(ecnt, elayout);
326  i_ecnt = ix % padded_ecnt;
327  i_esz = ix / padded_ecnt;
328  }
329  int i_ecnt_new = e_permute ? e_permute[i_ecnt] : i_ecnt;
330  int ix_new = nrn_i_layout(i_ecnt_new, ecnt, i_esz, esz, elayout);
331  *pd = ix_new + edata0;
332  } else {
333  nrn_assert(0);
334  }
335  }
336  } else if (s >= 0 && s < 1000) { // ion
337  int etype = s;
338  int elayout = corenrn.get_mech_data_layout()[etype];
339  Memb_list* eml = nt._ml_list[etype];
340  int edata0 = eml->data - nt._data;
341  int ecnt = eml->nodecount;
342  int esz = corenrn.get_prop_param_size()[etype];
343  int* e_permute = eml->_permute;
344  for (int iml = 0; iml < cnt; ++iml) {
345  int* pd = pdata + nrn_i_layout(iml, cnt, i, psz, layout);
346  int ix = *pd - edata0;
347  // from ix determine i_ecnt and i_esz (need to permute i_ecnt)
348  int i_ecnt, i_esz, padded_ecnt;
349  if (elayout == Layout::AoS) {
350  padded_ecnt = ecnt;
351  i_ecnt = ix / esz;
352  i_esz = ix % esz;
353  } else { // SoA
354  assert(elayout == Layout::SoA);
355  padded_ecnt = nrn_soa_padded_size(ecnt, elayout);
356  i_ecnt = ix % padded_ecnt;
357  i_esz = ix / padded_ecnt;
358  }
359  int i_ecnt_new = e_permute[i_ecnt];
360  int ix_new = nrn_i_layout(i_ecnt_new, ecnt, i_esz, esz, elayout);
361  *pd = ix_new + edata0;
362  }
363  }
364  }

429 static bool nrn_index_sort_cmp(const std::pair<int, int>& a, const std::pair<int, int>& b) {
430  bool result = false;
431  if (a.first < b.first) {
432  result = true;
433  } else if (a.first == b.first) {
434  if (a.second < b.second) {
435  result = true;
436  }
437  }
438  return result;
439 }
